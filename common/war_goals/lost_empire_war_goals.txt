wg_total_annihilation = {
	casus_belli = cb_total_annihilation

	hide_if_no_cb = no
	total_war = yes
	destroy_starbases = yes
	
	surrender_acceptance = -1000
	
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}
	
	set_defender_wargoal = wg_total_annihilation
	
	war_exhaustion = 0.25

	potential = {
		OR = {
			is_country_type = ascended_empire
			is_country_type = awakened_fallen_empire
		}
		OR = {
			from = {
				OR = {
					has_valid_civic = civic_fanatic_purifiers
					has_valid_civic = civic_hive_devouring_swarm
					has_valid_civic = civic_machine_assimilator
					has_valid_civic = civic_machine_terminator
				}
			}
			has_country_flag = rebel_lost_empire
			AND = {
				is_country_type = awakened_fallen_empire
				from = {
					has_country_flag = original_people
				}
			}
			AND = {
				has_country_flag = original_people
				from = {
					is_country_type = awakened_fallen_empire
				}
			}
			AND = {
				from = {
					OR = {
						is_country_type = ascended_empire
						is_fallen_empire = yes
					}
				}
				OR = {
					AND = {
						has_ethic = ethic_fanatic_spiritualist
						from = { has_ethic = ethic_fanatic_materialist }
					}
					AND = {
						has_ethic = ethic_fanatic_materialist
						from = { has_ethic = ethic_fanatic_spiritualist }
					}
					AND = {
						has_ethic = ethic_fanatic_xenophile
						from = { has_ethic = ethic_fanatic_xenophobe }
					}
					AND = {
						has_ethic = ethic_fanatic_xenophobe
						from = { has_ethic = ethic_fanatic_xenophile }
					}
					AND = {
						has_ethic = ethic_fanatic_militarist
						from = { has_ethic = ethic_fanatic_pacifist }
					}
					AND = {
						has_ethic = ethic_fanatic_pacifist
						from = { has_ethic = ethic_fanatic_militarist }
					}
					AND = {
						has_ethic = ethic_fanatic_authoritarian
						from = { has_ethic = ethic_fanatic_egalitarian }
					}
					AND = {
						has_ethic = ethic_fanatic_egalitarian
						from = { has_ethic = ethic_fanatic_authoritarian }
					}						
				}
			}
		}
	}
	
	possible = {
		custom_tooltip = {
			fail_text = "requires_terrible_opinion_or_supremacist"
			OR = {
				opinion_level = { who = from level <= terrible }
				is_harming_relations_with = from
				from = { is_at_war_with = prev }
				OR = {
					has_policy_flag = "diplo_stance_supremacist"
					from = { has_policy_flag = "diplo_stance_supremacist" }
					is_fallen_empire = yes
					from = { is_fallen_empire = yes }
				}
				from = { is_rival = prev }
			}
		}
	}
	
	on_accept  = {
			
		hidden_effect = {
			every_country = {
				if = {
					limit = {
						NOT = { is_same_value = root }
					}
					add_opinion_modifier = { modifier = opinion_genocidal who = root }
					add_threat = {
						who = root
						amount = 0.50
					}	
				}
			}
		}

	
#		
		country_event = { id = lost_empire.16 }
		remove_country_flag = ae_crusader
		save_event_target_as = Annihilator
		
		from = {
#			save_event_target_as = Annihilated
			
			country_event = { id = lost_empire.13 }
			remove_country_flag = ae_crusader
			
			every_subject = {

				country_event = { id = lost_empire.14 }
				
				every_owned_planet = {
					destroy_colony = yes
				}
			}
			every_owned_planet = {
				destroy_colony = yes
			}
			every_owned_fleet = {
				destroy_fleet = this
			}
		}
		hidden_effect = {
			every_playable_country = {
				limit = {
					NOR = {
						is_same_value = from
						is_same_value = root
					}
				}
				country_event = { id = lost_empire.10 }
			}	
		}

	}	
	
	on_status_quo = {
	
		hidden_effect = {
						
			every_country = {
				if = {
					limit = {
						NOT = { is_same_value = root }
					}
					add_opinion_modifier = { modifier = opinion_genocidal who = root }
					add_threat = {
						who = root
						amount = 0.50
					}	
				}
				if = {
					limit = {
						NOT = { is_same_value = from }
					}
					add_opinion_modifier = { modifier = opinion_genocidal who = from }
					add_threat = {
						who = from
						amount = 0.50
					}	
				}
			}
		}

		save_event_target_as = Annihilator
		every_planet_within_border = {
			limit = {
				is_owned_by = from
				is_colony = yes
			}
			destroy_colony = yes				
		}
		remove_country_flag = ae_crusader
		country_event = { id = lost_empire.17 }
			

		fromfrom = {
			save_event_target_as = Annihilated
			every_planet_within_border = {
				limit = {
					is_owned_by = root 
					is_colony = yes
				}
				destroy_colony = yes
			}
			remove_country_flag = ae_crusader
			country_event = { id = lost_empire.17 }
			
		}
		
		hidden_effect = {
		
			every_playable_country = {
				limit = {
					NOR = {
						is_same_value = root
						is_same_value = from
					}
				}
				country_event = { id = lost_empire.17 }
			}	
		}

		
		every_country = {
			if = {
				limit = { 	
					OR = {
#						is_same_value = root
						is_in_federation_with = root
						AND = {
							is_subject = yes
							exists = overlord
							overlord = { is_same_value = root }
						}
					}
				}
				every_planet_within_border = {
					limit = { is_owned_by = fromfrom }
					destroy_colony = yes
					hidden_effect = {
						solar_system = {
							every_fleet_in_system = {
								limit = {
									is_ship_class = shipclass_starbase
								}
								destroy_fleet = this
							}
						}
					}
				}
			}
			if = {
				limit = { 	
					OR = {
#						is_same_value = from
						is_in_federation_with = from
						AND = {
							is_subject = yes
							exists = overlord
							overlord = { is_same_value = from }
						}
					}
				}
				every_planet_within_border = {
					limit = { is_owned_by = root }
					destroy_colony = yes
					hidden_effect = {
						solar_system = {
							every_fleet_in_system = {
								limit = {
									is_ship_class = shipclass_starbase
								}
								destroy_fleet = this
							}
						}
					}
				}
			}	
		}
	}
	
	ai_weight = {
		weight = 200
	}
}

wg_banish_ascendancy = {
	casus_belli = cb_banish_ascendancy

	hide_if_no_cb = no
	release_occupied_systems_on_status_quo = yes
	total_war = no
	
	surrender_acceptance = -100
	
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}
	
	set_defender_wargoal = wg_banish_ascendancy
	
	war_exhaustion = 0.5

	potential = {
		OR = {
			is_country_type = ascended_empire
			is_country_type = awakened_fallen_empire
		}
		From = {
			OR = {
				is_country_type = ascended_empire
				is_country_type = awakened_fallen_empire
			}
		}
		NAND = {
			is_galactic_emperor = yes
			from = { is_galactic_community_member = yes }
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				OR = {
					is_country_type = ascended_empire
					is_country_type = awakened_fallen_empire
				}
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = cannot_subjugate_empire_member
			NAND = {
				has_galactic_emperor = yes
				is_galactic_community_member = no
				from = { is_galactic_community_member = yes }
			}
		}
		custom_tooltip = {
			fail_text = "cannot_use_against_custodian"
			from = { is_galactic_custodian = no }
		}
		custom_tooltip = {
			fail_text = "cannot_use_against_emperor"
			from = { is_galactic_emperor = no }
		}
		custom_tooltip = {
			fail_text = "cannot_use_against_btc"
			from = { has_been_declared_crisis = no }
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
		custom_tooltip = {
			fail_text = "requires_poor_or_worse_opinion"
			OR = {
				opinion_level = { who = from level <= poor }
				is_harming_relations_with = from
			}
		}
	}
	
	on_accept  = {
		hidden_effect = {
			from = {
				add_threat = {
					who = root
					amount = 1.0
				}
			}
			save_event_target_as = Banisher
			event_target:Banisher = {
				country_event = { id = lost_empire.15 }
				remove_country_flag = ae_crusader
			}
			from = {
				save_event_target_as = Banished
				remove_country_flag = ae_crusader
				every_subject = {
					set_subject_of = {
						who = root
						subject_type = vassal
					}
				}
				set_subject_of = {
					who = event_target:Banisher
					subject_type = banished_ascendancy
				}				
				country_event = { id = lost_empire.1002 }
			}
			every_playable_country = {
				limit = {
					NOT = {
						is_same_value = root
						is_same_value = from
					}
				}
				country_event = { id = lost_empire.11 }
			}	
		}
	}	
	
	on_status_quo = {
		remove_country_flag = ae_crusader
		if = {
			limit = {
				exists = last_created_country
				last_created_country = {
					has_country_flag = released_empire
				}
			}

			last_created_country = {
				set_subject_of = {
					who = root
					subject_type = vassal
				}
			}
		}
		from = {
			remove_country_flag = ae_crusader
#			every_subject = {
#				set_subject_of = {
#					who = none
#				}
#			}
		}
	}
	
	ai_weight = {
		weight = 100
	}
}
