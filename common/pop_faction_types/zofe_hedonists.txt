##############################
# Peaceful Hedonists Faction
##############################

hedonist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_pacifist
	use_guiding_ethic_as_pop_filter = no #this means that the faction will let pops that aren't egalitarian in too

	unique = yes
	support_multiplier = 1

	is_potential = {
		is_zofe_country = yes
		NOR = {
			has_ethic = ethic_gestalt_consciousness
			has_country_flag = suppress_vanilla_factions
		}
		count_owned_pop = {
			count >= 5
			limit = {
				has_job = ae_hedonist
			}
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	resources = {
		category = pop_factions
		produces = {
			influence = 2
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
		has_job = ae_hedonist
	}

	attraction = {
		base = 10000

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}

		modifier = {
			factor = 1.5
			is_pacifist = yes
		}

		# Should really check for citizenship instead
		modifier = {
			factor = 1.25
			NOT = { is_same_species = owner_species }
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			OR = {
				leader_class = admiral
				leader_class = general
			}
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	demand = {
		title = "HEDONIST_UTOPIAN"
		unfulfilled_title = "HEDONIST_NO_UTOPIAN"
		desc = "HEDONIST_UTOPIAN_DESC"

		unfulfilled_effect = -25
		fulfilled_effect = 25

		potential = {
			exists = owner
		}
		
		trigger = {
			owner = {
				any_owned_pop = {
					is_same_species = owner.species
					OR = {
						has_living_standard = { type = living_standard_utopian }
						has_living_standard = { type = living_standard_decadent }
					}
				}
			}
		}
	}

	demand = {
		title = "PROSPERITY_PHILOSOPHY"
		unfulfilled_title = "PROSPERITY_NO_PHILOSOPHY"
		desc = "PROSPERITY_PHILOSOPHY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				any_relation = {
					has_communications = root.owner
					is_zofe_compatible = yes
					NOT = { exists = overlord }
				}
			}
		}

		trigger = {
			owner = {
				has_policy_flag = no_wars
			}
		}
	}

	demand = {
		title = "PROSPERITY_PEACE"
		unfulfilled_title = "PROSPERITY_NO_PEACE"
		desc = "PROSPERITY_PEACE_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				is_at_war = no
			}
		}
	}
	
	demand = {
		title = "PROGRESSIVE_REPRODUCTIVE_FREEDOM"
		unfulfilled_title = "PROGRESSIVE_NO_REPRODUCTIVE_FREEDOM"
		desc = "PROGRESSIVE_REPRODUCTIVE_FREEDOM_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			NOT = {
				owner = { has_policy_flag = population_controls_allowed }
			}
		}
	}

	on_create = {
		pop_faction_event = { id = zofe_factions.1 }
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_hedonist
			}
			remove_modifier = promoted_hedonist
		}
		if = {
			limit = {
				has_modifier = suppressed_hedonist
			}
			remove_modifier = suppressed_hedonist
		}
	}
	
	
	

	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"

			cost = {
				influence = 500
			}

			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}

			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_pacifist" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_pacifist
					hidden_effect = {
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_zofe_compatible = yes
								has_communications = prev
								has_intel_level = {
									who = prev
									category = government
									level >= 1
								}
							}
							country_event = { id = factions.2000 }
						}
					}
				}
			}

			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_pacifist
					}
				}
				modifier = {
					factor = 0
					owner = { has_valid_civic = civic_fanatic_purifiers }
				}
				modifier = {
					factor = 0
					NOT = {
						owner = {
							exists = ruler
							ruler = {
								leader_of_faction = prosperity
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_faction = {
							count < 4
						}
					}
				}
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_hedonist
						has_modifier = promoted_hedonist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = promoted_hedonist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_hedonist
				}
			}

			effect = {
				parameter:empire = {
					remove_modifier = promoted_hedonist
				}
			}

			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_hedonist
						has_modifier = promoted_hedonist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_hedonist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_hedonist
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_hedonist
				}
			}

			ai_weight = {
				base = 0
			}
		}
	}
}
