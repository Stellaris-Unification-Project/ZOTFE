building_fe_affluence_center = {
	base_buildtime = @b2_time
	can_build = yes
	can_demolish = yes
	icon = building_hyper_entertainment_forum
#	base_cap_amount = 1
	
	category = manufacturing
	
	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
#			buildings_no_crystals = yes
			
			AND = {
				exists = owner
				owner = {
					is_ai = yes
					country_uses_consumer_goods = no
				}
			}
		}
	}
	
	potential = {
		exists = owner
		owner = {
			is_zofe_country = yes
			country_uses_consumer_goods = yes
		}
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		building_relaxed_basic_income_check = yes
		OR = {
			NOT = { is_planet_class = pc_habitat }
			has_planet_flag = advanced_habitat_2
		}
	}
	
	allow = {
		has_upgraded_capital = yes
#		buildings_has_crystals = yes
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b3_minerals
		}
		upkeep = {
			minerals = 25
			energy = 10
		}
		produces = {
			consumer_goods = 50
		}
		produces = {
			trigger = {
				exists = owner
				owner = {
					has_edict = planet_fe_buildings_boost
				}	
			}
			consumer_goods = 10
		}
		produces = {
			trigger = {
				planet = {
					has_active_building = building_fe_automated_production
				}	
			}
			consumer_goods = 15
		}
	}
		
	upgrades = {
		building_fe_opulent_plaza
	}
	
	prerequisites = {
		"tech_fe_affluence_center"
	}	
}

building_fe_sky_dome = {
	base_buildtime = @b2_time
	can_build = yes
	icon = building_paradise_dome
#	base_cap_amount = 1
	
	category = amenity
	
	destroy_trigger = {
		exists = owner
		OR = {
			AND = {
				free_building_slots = 0
				free_jobs < 1
				free_housing > 10
				owner = {
					is_ai = yes
				}
			}
			owner = { is_regular_empire = no }
			owner = { has_valid_civic = civic_shared_burden }
			has_modifier = slave_colony
#			buildings_no_crystals = yes
		}
	}
	
	potential = {
		OR = {
			NOT = { is_planet_class = pc_habitat }
			AND = {
				has_planet_flag = advanced_habitat
				exists = owner
				owner = {
					has_ascension_perk = ap_voidborn
				}
			}
		}
		NOT = { has_modifier = slave_colony }
		exists = owner
		owner = {
			is_zofe_country = yes
			is_regular_empire = yes
			NOT = { has_valid_civic = civic_shared_burden }
		}
	}
	
	allow = {
		has_upgraded_capital = yes
		buildings_simple_allow = yes
#		buildings_has_crystals = yes
	}
	
	convert_to = {
		building_organic_paradise
		building_drone_megastorage
		building_expanded_warren
		building_communal_housing_large
	}
	
	planet_modifier = {

		planet_amenities_add = 20
	}
	
	triggered_planet_modifier = {

			exists = owner

			energy = 10
		}

	upgrades = {
	}
	
	prerequisites = {
		"tech_fe_sky_dome"
	}
	
	show_tech_unlock_if = {
		owner = {
			is_regular_empire = yes
			NOT = { has_valid_civic = civic_shared_burden }
		}
	}
}
